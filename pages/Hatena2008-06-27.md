>|python|
>>> class Foo(object):
...     def __init__(self, name):
...             self.name = name
...     def __del__(self):
...             print self.name, "deleted"
... 
>>> x = Foo("x")
>>> y = Foo("y")
>>> x = y = 1
x deleted
y deleted
