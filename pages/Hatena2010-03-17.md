from operator import mul
def perm(n, m):
    divs = [reduce(mul, [n - i for i in range(j)], 1) for j in range(m + 1)]
    total = divs[-1]
    for x in xrange(total):
        yield [x % divs[i + 1] / divs[i] for i in range(m)]
