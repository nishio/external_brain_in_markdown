	var b:Number = end - G2 * diff;

data Pair = Pair {car::Pair, cdr::Pair} | Null | Value Int

instance Show Pair where
    show (Pair x y) = "(" ++ show x ++ " . " ++ show y ++ ")"
    show Null = "()"
    show (Value x) = show x


main = putStrLn $ show $ (Pair (Value 1) (Pair (Value 2) Null))

instance Show Foo where ...
